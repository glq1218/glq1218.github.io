安装 #

ubuntu #

卸载旧版本 #

bash
sudo apt-get remove docker docker-engine io containerd runc

设置存储库 #

更新APT程序包索引并安装程序包，以允许APT通过HTTPS使用存储库： #
bash
sudo apt-get update
sudo apt-get install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release
添加Docker官方GPG密钥：(这里使用了阿里云镜像源，官方源：download.com，国内访问慢) #
bash
sudo mkdir -p /etc/apt/keyrings
curl -fsSL https://mirrors.aliyun.com/docker-ce/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/gpg
设置存储库： #
bash
echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/gpg] https://mirrors.aliyun.com/docker-ce/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/list > /dev/null

安装Docker引擎 #

更新APT包索引： #
bash
sudo apt-get update
安装Docker Engine、Containerd和Docker Compose。 #
bash
sudo apt-get install docker-ce docker-ce-cli containerd.io docker-compose-plugin
通过运行Hello-world映像验证Docker Engine安装是否成功： #
bash
sudo docker run hello-world

阿里云镜像加速器 #

bash
sudo mkdir -p /etc/docker
sudo tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://sj3v78kt.mirror.aliyuncs.com"]
}
EOF
sudo systemctl daemon-reload
sudo systemctl restart docker

mysql 主从复制 #

创建主服务器容器实例 3307 #

bash
docker run -p 3307:3306 --privileged=true -v /root/docker/mysql/mysql-master/log:/var/log/mysql -v /root/docker/mysql/mysql-master/data:/var/lib/mysql -v /root/docker/mysql/mysql-master/conf:/etc/mysql/conf.d -e MYSQL_ROOT_PASSWORD=4265474264@Root --name mysql-master -d mysql

创建配置文件 #

yaml
[client]
# 设置客户端编码
default_character_set = utf8mb4
[mysqld]
## 设置编码
collation_server = utf8mb4_general_ci
character_set_server = utf8mb4
## 设置server_id，同一局域网中需要唯一
server_id=101
## 指定不需要同步的数据库名称
binlog-ignore-db=mysql
## 开启二进制日志功能
log-bin=mall-mysql-bin
## 设置二进制日志使用内存大小（事务）
binlog_cache_size=1M
## 设置使用的二进制日志格式（mixed,statement,row）
binlog_format=mixed
## 二进制日志过期清理时间。默认值为0，表示不自动清理。
expire_logs_days=7
## 跳过主从复制中遇到的所有错误或指定类型的错误，避免slave端复制中断。
## 如：1062错误是指一些主键重复，1032错误是因为主从数据库数据不一致
slave_skip_errors=1062

添加用户授予权限 #

sql
create user 'slave'@'%' identified by '4265474264@Root';
grant replication slave, replication client on *.* to 'slave'@'%';

创建从服务器容器实例 3308 #

bash
docker run -p 3308:3306 --privileged=true -v /root/docker/mysql/mysql-slave/log:/var/log/mysql -v /root/docker/mysql/mysql-slave/data:/var/lib/mysql -v /root/docker/mysql/mysql-slave/conf:/etc/mysql/conf.d -e MYSQL_ROOT_PASSWORD=4265474264@Root --name mysql-slave -d mysql

创建配置文件 #

yaml
[client]
# 设置客户端编码
default_character_set = utf8mb4
[mysqld]
## 设置编码
collation_server = utf8mb4_general_ci
character_set_server = utf8mb4
## 设置server_id，同一局域网中需要唯一
server_id=102
## 指定不需要同步的数据库名称
binlog-ignore-db=mysql
## 开启二进制日志功能，以备Slave作为其它数据库实例的Master时使用
log-bin=mall-mysql-slave1-bin
## 设置二进制日志使用内存大小（事务）
binlog_cache_size=1M
## 设置使用的二进制日志格式（mixed,statement,row）
binlog_format=mixed
## 二进制日志过期清理时间。默认值为0，表示不自动清理。
expire_logs_days=7
## 跳过主从复制中遇到的所有错误或指定类型的错误，避免slave端复制中断。
## 如：1062错误是指一些主键重复，1032错误是因为主从数据库数据不一致
slave_skip_errors=1062
## relay_log配置中继日志
relay_log=mall-mysql-relay-bin
## log_slave_updates表示slave将复制事件写进自己的二进制日志
log_slave_updates=1
## slave设置为只读（具有super权限的用户除外）
read_only=1

3主3从 redis 集群配置 #

启动六台 redis #

bash
docker run -d --name redis-node-1 --net host --privileged=true -v /root/docker/redis/redis-node-1:/data redis --cluster-enabled yes --appendonly yes --port 6381
docker run -d --name redis-node-2 --net host --privileged=true -v /root/docker/redis/redis-node-2:/data redis --cluster-enabled yes --appendonly yes --port 6382
docker run -d --name redis-node-3 --net host --privileged=true -v /root/docker/redis/redis-node-3:/data redis --cluster-enabled yes --appendonly yes --port 6383
docker run -d --name redis-node-4 --net host --privileged=true -v /root/docker/redis/redis-node-4:/data redis --cluster-enabled yes --appendonly yes --port 6384
docker run -d --name redis-node-5 --net host --privileged=true -v /root/docker/redis/redis-node-5:/data redis --cluster-enabled yes --appendonly yes --port 6385
docker run -d --name redis-node-6 --net host --privileged=true -v /root/docker/redis/redis-node-6:/data redis --cluster-enabled yes --appendonly yes --port 6386

构建主从关系 #

bash
redis-cli --cluster create 101.35.46.21:6381 101.35.46.21:6382 101.35.46.21:6383 101.35.46.21:6384 101.35.46.21:6385 101.35.46.21:6386 --cluster-replicas 1

查看集群状态 #

bash
redis-cli -p 6381

cluster info
cluster nodes
# -c 集群方式启动
redis-cli -p 6381 -c
# 查看集群情况
redis-cli --cluster check 101.35.46.21:6381

redis主从扩容 #

创建两个redis节点 #

bash
docker run -d --name redis-node-7 --net host --privileged=true -v /root/docker/redis/redis-node-7:/data redis --cluster-enabled yes --appendonly yes --port 6387
docker run -d --name redis-node-8 --net host --privileged=true -v /root/docker/redis/redis-node-8:/data redis --cluster-enabled yes --appendonly yes --port 6388

新增6387作为master节点 #

bash
redis-cli --cluster add-node 101.35.46.21:6387 101.35.46.21:6381
6387需要作为新增master的节点
6381原来集群节点里面的领路人

检查集群情况第一次 #

bash
redis-cli --cluster check 101.35.46.21:6381

重新分配槽号 #

bash
redis-cli --cluster reshard 101.35.46.21:6381

检查集群情况第二次 #

bash
redis-cli --cluster check 101.35.46.21:6381

为6387分配从节点6388 #

bash
redis-cli --cluster add-node 101.35.46.21:6388 101.35.46.21:6387 --cluster-slave --cluster-master-id b25d5c006642c7735863f6b3a78552ee563508d2

6387节点的id

检查集群情况第三次 #

bash
redis-cli --cluster check 101.35.46.21:6381

redis 主从缩容 #

检查集群情况获取6388的节点id #

bash
redis-cli --cluster check 101.35.46.21:6381

删除从节点6388 #

bash
redis-cli --cluster del-node 101.35.46.21:6388 552e947031be03584cb5f171a9632c09a0e03c9f

将6387的槽号清空，重新分配，本例将清空出来的槽号都给6381 #

bash
redis-cli --cluster reshard 101.35.46.21:6381

删除6387主节点 #

bash
redis-cli --cluster del-node 101.35.46.21:6387 b25d5c006642c7735863f6b3a78552ee563508d2
